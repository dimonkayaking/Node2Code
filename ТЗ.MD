```csharp
public abstract class VisualScriptingNode : Node
{
    public string NodeId;           // ID из GraphData
    public NodeType LogicType;       // Тип ноды
    
    public virtual void InitializeFromData(NodeData data)
    {
        NodeId = data.Id;
        LogicType = data.Type;
        // установить позицию и т.д.
    }
    
    public virtual NodeData ToNodeData()
    {
        return new NodeData
        {
            Id = NodeId,
            Type = LogicType,
            // заполнить из портов
        };
    }
}
```

### 4. **Создать GraphSerializer**
```csharp
public static class GraphSerializer
{
    // NodeGraph → GraphData (для генерации кода)
    public static GraphData ToGraphData(VisualScriptingGraph graph)
    {
        var graphData = new GraphData();
        // обойти все ноды, собрать данные
        // обойти все соединения, собрать связи
        return graphData;
    }
    
    // GraphData → NodeGraph (для отрисовки)
    public static void FromGraphData(VisualScriptingGraph graph, GraphData data)
    {
        graph.Clear();
        foreach (var nodeData in data.Nodes)
        {
            var node = CreateNodeFromType(nodeData.Type);
            node.InitializeFromData(nodeData);
            graph.AddNode(node);
        }
        // создать соединения из InputConnections и ExecutionFlow
    }
}
```

### 5. **Настроить порты для Execution Flow (белые провода)**

```csharp
[Node("Flow/If")]
public class IfNode : VisualScriptingNode
{
    [Input(name = "Condition")] public bool condition;
    [Input(name = "Exec In", showBackingValue = Never)] public ExecutionFlow execIn;
    [Output(name = "True", showBackingValue = Never)] public ExecutionFlow execTrue;
    [Output(name = "False", showBackingValue = Never)] public ExecutionFlow execFalse;
    [Output(name = "Exec Out", showBackingValue = Never)] public ExecutionFlow execOut;
}
```
